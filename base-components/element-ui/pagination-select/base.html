  • UHP450*2250（450*2250 UHP)
  • 二培UHP450*2250（450*2250 UHP)
  • 石墨UHP450*2250-6（450*2550 UHP)
  • 石墨UHP450*2250-7（450*2550 UHP)
  • 石墨UHP450*2250-8（450*2550 UHP)
  • 石墨UHP450*2250-9（450*2550 UHP)

无数据

在组件中需配置：

@pageChange：选择分页的事件；传回当前选中的页码
@selectChange：下拉选择事件；传回当前选中的值

optionSource 下拉框组件数据源
valueKey 传入的 option 数组中，要作为最终选择项的键值名称
labelKey 传入的 option 数组中，要作为显示项的键值名称
paginationOption object 默认值
whetherSetAllSelect 是否需要新增全部选项 ； boolean值

pageSize: 6,//每页显示条数
currentPage: 1,//当前页
pagerCount: 5,//按钮数，超过时会折叠
total: 0 //总条数

<template>
  <div
    class="t-pagination-select-demo"
    style="width: 100%; min-height: 100px; padding: 15px"
  >
    <el-form
      :model="reactiveData.addForm"
      :rules="reactiveData.addFormRules"
      ref="addForm"
      label-width="100px"
      class="demo-ruleForm"
    >
      <el-form-item label="品名(原料)" prop="materialId">
        <rt-pagination-select
          @pageChange="pageChange"
          @selectChange="selectChange"
          :optionSource="reactiveData.materialIdArr"
          v-model="reactiveData.addForm.materialId"
          labelKey="materialName"
          valueKey="id"
          :paginationOption="reactiveData.setSelectPage"
          whetherSetAllSelect
        />
      </el-form-item>
    </el-form>
    <el-button type="primary" @click="submitImportForm">提交</el-button>
  </div>
</template>
<script setup>
import data from "./data.json";
import data2 from "./data2.json";
import { reactive ,ref} from "vue";
const reactiveData = reactive({
  addForm: {
    materialId: null, // 产出品名
  },
  addFormRules: {
    materialId: [
      { required: true, message: "请选择品名(原料)！", trigger: "change" },
    ],
  },
  materialIdArr: [],
  setSelectPage: {
    pageSize: 6, // 每页显示条数
    currentPage: 1, // 当前页
    pagerCount: 7, // 按钮数，超过时会折叠
    total: 0, // 总条数
  },
});
const addForm = ref(null);
// 获取品名下拉数据
const selectMaterialList = (pageNum) => {
  let res;
  if (pageNum === 1) {
    res = data;
  } else {
    res = data2;
  }
  if (res.success) {
    // console.log('获取品名下拉数据', res.data)
    reactiveData.materialIdArr = res.data.records;
    reactiveData.setSelectPage.total = res.data.total;
  }
};
selectMaterialList(1)

// 下拉选择分页
const pageChange = (val) => {
  reactiveData.setSelectPage.currentPage = val;
  selectMaterialList(reactiveData.setSelectPage.currentPage);
};

// 回显的值
const selectChange = (val)=>{
  console.log("回显的值为："+val)
}

const submitImportForm = () => {
  addForm.value && addForm.value.validate((valid) => {
    if (!valid) return;
    console.log("提交", reactiveData.addForm);
  });
};
</script>
