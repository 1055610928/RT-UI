<template>
  <div class="t-date-picker-demo" style="min-height:100px;width:100%;padding:10px;">
    <div>基本模式</div>
    <div style="display:flex;">
      <rt-date-picker 
        :dateVal="[reactiveData.formDate.warnStartDate,reactiveData.formDate.warnEndDate]" 
        @startChange="startChange"
        @endChange="endChange"
      ></rt-date-picker>
      <rt-date-picker 
        style="margin-left:15px;" 
        :startDate="reactiveData.formDate.warnStartDate" 
        :endDate="reactiveData.formDate.warnEndDate"
        dispaysType="two" 
        @startChange="startChange"
        @endChange="endChange"
      ></rt-date-picker>
    </div>

    <div>含时分秒：需要配置plusTime</div>
    <div style="display:flex;">
      <rt-date-picker 
        plusTime 
        :dateVal="[reactiveData.warnStartDate,reactiveData.warnEndDate]" 
        @startChange="startChange1"
        @endChange="endChange2"
      ></rt-date-picker>
      <rt-date-picker 
        plusTime 
        style="margin-left:15px;" 
        dispaysType="two" 
        :startDate="reactiveData.warnStartDate"
        :endDate="reactiveData.warnEndDate" 
        @startChange="startChange1" 
        @endChange="endChange2"
      ></rt-date-picker>
    </div>
    <div>默认显示日期</div>
    <div style="display:flex;">
      <rt-date-picker 
        :dateVal="[reactiveData.startDate,reactiveData.endDate]" 
        @startChange="startChange3" 
        @endChange="endChange4"
      />
      <rt-date-picker 
        style="margin-left:15px;" 
        dispaysType="two" 
        :startDate="reactiveData.startDate" 
        :endDate="reactiveData.endDate"
        @startChange="startChange3" 
        @endChange="endChange4"
      ></rt-date-picker>
    </div>
  </div>
</template>

<script setup lang="ts">
import { reactive } from 'vue'
 function dateFormatFilter (time:Date) {
    let date: Date|string = new Date(time)
    let year = date.getFullYear()
    let month: number| string = date.getMonth() + 1
    let day : number| string= date.getDate()
    month = month < 10 ? '0' + month : month
    day = day < 10 ? '0' + day : day
    date = year + '-' + month + '-' + day
    return date
}
const reactiveData = reactive({
  formDate: {
    warnStartDate: '',
    warnEndDate: ''
  },
  warnStartDate: '',
  warnEndDate: '',
  startDate: dateFormatFilter(new Date((new Date().getTime() - 3600 * 1000 * 24 * 90))),
  endDate: dateFormatFilter(new Date())
})
// 开始时间
const startChange = (val:string) =>{
  console.log('开始时间', val)
  reactiveData.formDate.warnStartDate = val
}
// 结束时间
const endChange = (val:string) => {
  console.log('结束时间', val)
  reactiveData.formDate.warnEndDate = val
}
// 开始时间
const startChange1 = (val:string) => {
  console.log('开始时间', val)
  reactiveData.warnStartDate = val
}
// 结束时间
const endChange2 = (val:string) => {
  console.log('结束时间', val)
  reactiveData.warnEndDate = val
}
// 开始时间
const startChange3 = (val:string) => {
  console.log('开始时间', val)
  reactiveData.startDate = val
}
// 结束时间
const endChange4 = (val:string) => {
  console.log('结束时间', val)
  reactiveData.endDate = val
}
</script>
